from dateutil.parser import parse
from datetime import datetime
data['Datetime'] = data.Date.astype(str).str.cat(data.Time.astype(str), sep=' ')
datetime=[]
for i in data['Datetime']:
    datetime.append(parse(i))
se=pd.Series(datetime)
data['DatetimeFormat']=se.values

important_crime_categories = ['LARCENY/THEFT','OTHER OFFENSES', 'NON-CRIMINAL', 'ASSAULT', 'VEHICLE THEFT',
                       'WARRANTS','DRUG/NARCOTIC', 'SUSPICIOUS OCC','MISSING PERSON']
for category in important_crime_categories:
    df = data.loc[data["Category"]==category]
    time_counts = df["DatetimeFormat"].value_counts()
    plt.figure(figsize = (10,5)); time_counts.plot(legend=True, title=category); plt.show()
